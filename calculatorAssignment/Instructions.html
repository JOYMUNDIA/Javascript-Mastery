JS Panel

let display = document.getElementById('display');
let currentInput = '';
let expression = '';
let firstNumber = null;
let operator = null;
let shouldResetDisplay = false;

function appendNumber(num) {
  if (shouldResetDisplay) { currentInput = ''; shouldResetDisplay = false; }
  if (num === '.' && currentInput.includes('.')) return;
  currentInput += num;
  updateDisplay();
}

function appendOperator(op) {
  if (currentInput === '') return;
  if (firstNumber === null) {
    firstNumber = parseFloat(currentInput);
    expression = currentInput + ' ' + op + ' ';
  } else if (operator) {
    firstNumber = operate(firstNumber, parseFloat(currentInput), operator);
    expression += currentInput + ' ' + op + ' ';
  }
  operator = op;
  shouldResetDisplay = true;
  updateDisplay(expression);
}

function clearDisplay() {
  currentInput = '';
  expression = '';
  firstNumber = null;
  operator = null;
  shouldResetDisplay = false;
  updateDisplay('0');
}

function deleteLast() {
  if (currentInput.length > 0) {
    currentInput = currentInput.slice(0, -1);
    updateDisplay(currentInput);
  }
}

function calculateResult() {
  if (firstNumber === null || operator === null || currentInput === '') return;
  let secondNumber = parseFloat(currentInput);
  let result = operate(firstNumber, secondNumber, operator);
  expression += currentInput + ' = ' + result;
  updateDisplay(expression);
  currentInput = result.toString();
  firstNumber = null;
  operator = null;
  expression = '';
  shouldResetDisplay = true;
}
