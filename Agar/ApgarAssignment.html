const steps = document.querySelectorAll(".step");

function showStep(i) {
  steps.forEach(s => s.classList.remove("active"));
  steps[i].classList.add("active");
}

   CALCULATE APGAR SCORE
   STUDENT TASK: Observe how total score is computed
   - Loops through appearance, pulse, grimace, activity, respiration
   - Adds selected values
   - Updates the score display and result text with classes

function calculate() {
  let total = 0;

  ["appearance","pulse","grimace","activity","respiration"].forEach(name => {
    let selected = document.querySelector(`input[name="${name}"]:checked`);
    if (selected) total += parseInt(selected.value);
  });

  document.getElementById("score").textContent = total;

  let resultText = document.getElementById("resultText");

  /* =========================
   INTERPRETING APGAR SCORE
   STUDENT TASK: Display result text and style based on the total score
   - total is already calculated from the selected options
   - Decide what message to show for high, medium, and low scores
   - Assign the appropriate CSS class to style the result
========================= */

// 1. Check if total score is 7 or higher
//    - If yes, set the result text to indicate "Healthy (Normal)"

//    - If yes, set the result text to indicate "Moderate distress"
//    - Assign the CSS class for a warning/medium result

// 3. Otherwise (total score is below 4)
//    - Set the result text to indicate "Severe distress (Emergency)"
//    - Assign the CSS class for a critical/severe result

  currentStep = steps.length - 1; // go to result step
  showStep(currentStep);
}

/* =========================
   RESTART FUNCTION
   STUDENT TASK: See how restart resets the app
   - Clears all selections
   - Resets step to the beginning
========================= */

function restart() {
  document.querySelectorAll("input").forEach(i => i.checked = false);
  currentStep = 0;
  showStep(currentStep);
}
